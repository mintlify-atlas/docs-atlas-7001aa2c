    ```astro

        type="date"
    ```

    - Alimentación de animales
    - Foto con animales
    - Tour guiado privado

    - Tarjetas de crédito/débito
    - Transferencia bancaria
    - Pagos móviles (Orange Money, Tigo Money)
  </Step>
  
  <Step title="Confirmation & E-tickets">
    Generate QR code tickets and send confirmation email
  </Step>
</Steps>

```astro src/components/ZooMap.astro
interface MapPoint {
  id: string;
  name: string;
  type: 'exhibit' | 'facility' | 'restaurant' | 'restroom';

  <svg viewBox="0 0 800 600" class="w-100">
    <!-- Map background -->
    <image href="/images/zoo-map-base.svg" width="800" height="600" />
    
    <!-- Interactive points -->
    {mapPoints.map(point => (
      <g class="map-point" data-type={point.type}>
        <circle 
          cx={point.x} 
          cy={point.y} 
          r="10" 
          class="point-marker"
          data-bs-toggle="tooltip"

        <text x={point.x} y={point.y + 25} text-anchor="middle" class="point-label">
```

## Group Bookings

Special system for schools and large groups:

```typescript
interface GroupBooking {
  organizationName: string;
  contactPerson: string;
  groupSize: number;
  groupType: 'school' | 'corporate' | 'social';
  requestedDate: Date;
  alternativeDates: Date[];
  specialRequirements: string;
  educationalProgram: boolean;
}
```

**Group Benefits:**
- Discounted rates for 15+ people
- Dedicated guide included
- Private areas for lunch
- Customizable educational programs
- Flexible payment terms
